## Data Structure

### TableFieldData

```tsx
interface TableFieldData {
  columns: TableFieldColumn[]
  rows: TableFieldRow[]
}

interface TableFieldColumn {
  id: string
  label: string
  type?: "text" | "number" | "date" | "boolean" | "select"
  options?: Array<{ value: string; label: string }>
  required?: boolean
  defaultValue?: any
}

interface TableFieldRow {
  id: string
  data: Record<string, any>
}
```
